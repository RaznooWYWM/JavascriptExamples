// iterative for loop
      for (let i = 0; i < 10; i++){
         document.write(i); // will iteratively output all numbers from 0 to 9 inclusive
      }
---------------------------------------      
      // for each loop
      var my_array = [1, 3, 5, 6, 4 ,2];
      my_array.forEach(i =>document.write(i)); // will print all elements in my_array starting from 0th index
---------------------------------------      
      // while loop
      var i = 0;
      while (i  < 10){
         document.write(i); //prints out all numbers from 0 to 9 inclusive
         i = i + 1;
      }
